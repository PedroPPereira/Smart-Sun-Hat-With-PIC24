{

do{
scanf("%s", s);  // wait for a new character
printf("%s\n",s); // echo character
if ( *s=='\n') // line feed, ignore it
continue;
if ( *s=='\r') // end of line, end loop
break;

len--;
} while ( len>0 ); // until buffer full
